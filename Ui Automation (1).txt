        // Navigation
        function showSection(sectionId) {
            // Hide all sections
            document.querySelectorAll('.section').forEach(section => {
                section.classList.remove('active');
            });
            
            // Remove active from all tabs
            document.querySelectorAll('.nav-tab').forEach(tab => {
                tab.classList.remove('active');
            });

            document.getElementById(sectionId).classList.add('active');

        function startPipeline() {
            pipelineRunning = true;
            document.querySelectorAll('.stage-node').forEach((node, index) => {
                setTimeout(() => {
                    node.classList.add('active');
                }, index * 500);
            });
            addLogEntry('SUCCESS', 'Pipeline started successfully');
            updateStatus('PROCESSING');
        }

        function stopPipeline() {
            pipelineRunning = false;
            document.querySelectorAll('.stage-node').forEach(node => {
                node.classList.remove('active', 'processing');
            });
            addLogEntry('WARNING', 'Pipeline stopped by user');
            updateStatus('STOPPED');
        }

            stopPipeline();

                document.querySelectorAll('.stage-node').forEach(node => {

        }

        // Terminal/Logs
        function addLogEntry(type, message) {
            const terminal = document.getElementById('systemTerminal');
            const time = new Date().toLocaleTimeString('en-US', { hour12: false });
            const colors = {
                'SUCCESS': 'log-success',
                'INFO': 'log-info',
                'WARNING': 'log-warning',
                'ERROR': 'log-error'
            };
            
            const entry = document.createElement('div');
            entry.className = 'log-entry';
            entry.innerHTML = `
                <span class="log-time">[${time}]</span>
                <span class="${colors[type]}">[${type}]</span>
                <span>${message}</span>
            `;
            
            terminal.insertBefore(entry, terminal.firstChild);
            
            // Keep only last 50 entries
            while (terminal.children.length > 50) {
                terminal.removeChild(terminal.lastChild);
            }
        }
